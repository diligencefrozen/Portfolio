## 0. JavaScript 기본 이벤트

- **JavaScript**: 인터프리터 방식 언어
    
    → 위에서 아래로 한 줄씩 번역·실행
    
- **onload**: 페이지 로딩 완료 후 함수 실행
- **onclick**: 클릭 이벤트 발생 시 함수 실행
- **DOM 객체 가져오기**
    
    ```jsx
    document.getElementById("id값")
    ```
    
- input 값 가져오기
    
    ```jsx
    input객체.value
    ```
    

### 전체 코드

```html
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>Insert title here</title>
<script type="text/javascript">
	// 인터프리터 방식 언어:
	// 실행하면 위에서 한 줄씩 기계어로 번역해서 실행
	// cf) 컴파일 방식 언어: 전체를 한 번에 번역해서 실행

	// DOM 객체에 onXXX 이벤트를 연결
	let iInput = null;

	function init() {
		iInput = document.getElementById("i");
		alert(iInput);
	}

	function buttonInput() {
		alert(iInput.value);
	}

	// <button>과 buttonInput() 함수 연결
	// 버튼을 한 번 클릭했을 때 onclick 발생
	// alert 창에 input에 입력한 값 출력

	// 유효성 검사 vs XXXValidChecker.js
	// 유효성 검사를 편하게 하기 위해 함수를 정리해놓은 파일

	// 웹페이지 이벤트 처리 vs jQuery.js
	// JS 자체를 편하게 사용할 수 있도록 함수를 정리해놓은 파일
</script>
</head>
<body onload="init();">
	<!-- onload: 페이지가 모두 로딩된 후 init() 호출 -->
	<input id="i">
	<button onclick="buttonInput();">Print</button>
</body>
</html>
```

# 1. jQuery.html

```jsx
$(선택자).이벤트(콜백함수);
```

- **선택자**: CSS 선택자 사용
- **콜백함수**: 이벤트가 발생하면 자동 호출되는 함수
- 문서 준비 완료 후 실행:

```jsx
$(document).ready(function() {
	// 실행 내용
});
```

또는 축약형:

```jsx
$(function() {
	// 실행 내용
});
```

- jQuery input 값 가져오기:

```jsx
$("#i").val();
```

### 전체 코드

```html
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>Insert title here</title>
<script type="text/javascript" src="jQuery.js"></script>
<script type="text/javascript">
	// $(선택자).이벤트(콜백함수);
	// 선택자: CSS 선택자 이용
	// 콜백함수: 어떤 이벤트에 의해 호출되어지는 함수

	// 문서 준비가 끝나면 자동 호출
	/*
	$(document).ready(function init() {
		alert("준비 완료!");
	});
	*/

	/*
	$(function() {
		alert("준비 완료!");
	});
	*/

	$(function() {
		let b = false;

		$("h1").click(function() {
			alert("h1 클릭이벤트 발생!");
		});

		$("h2").click(() => {
			alert("h2 클릭이벤트 발생!");
			// 스위치 기능에서 사용할 변수는 바깥에 선언
		});

		$("h3").click(function() {
			b = !b;
			alert(b);
		});

		let iInput = $("#i");

		// html에 있는 button에 클릭 이벤트가 발생하면
		// 콜백함수가 자동으로 호출되도록 연결
		$("button").click(function() {
			alert(iInput.val());
		});

		// 이벤트 하나에 함수를 여러 개 연결 가능
	});
</script>
</head>
<body>
	<h1>다들 너무 피곤해보이시네요!</h1>
	<h2>와아아아 근데 아직 화요일이에요~</h2>
	<h3>주말까지 한참 남았어요 ^__^</h3>
	<hr>

	<input id="i">
	<button>출력!</button>
	<hr>
	<button>Print</button>
</body>
</html>
```

# 3. CSS.html — 움직이는 이미지 / 메뉴

### setInterval

```jsx
setInterval(() => {
	// 반복 실행할 코드
}, 시간);
```

- 일정 시간 간격으로 계속 반복 실행
- `1000` = 1초

### setTimeout

```jsx
setTimeout(() => {
	// 한 번만 실행할 코드
}, 시간);
```

### mouseenter / mouseleave

```jsx
$(선택자).mouseenter(function() {});
$(선택자).mouseleave(function() {});
```

- `mouseenter`: 마우스가 영역 안으로 들어올 때
- `mouseleave`: 마우스가 영역 밖으로 나갈 때

### attr

```jsx
$(선택자).attr("속성명", "값");
```

- 속성 가져오기 / 추가 / 변경


# 4. RepeatStmt.html

### JavaScript 자료형

```jsx
let arr = [1, 56, 3, 86, 123];          // 배열
let obj = { menu: "초밥", price: 15000 }; // 객체
```

- 배열: `[]`
- 객체: `{}`
- 배열 속 객체 형태 = JSON 표현에 자주 사용

### 배열 반복

```jsx
$.each(배열, function(index, data) {
	// 반복 실행
});
```

### DOM 객체 반복

```jsx
$(선택자).each(function(index, dom) {
	// 반복 실행
});
```

### 전체 코드

```html
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>Insert title here</title>
<script type="text/javascript" src="jQuery.js"></script>

<script type="text/javascript">
	$(function() {
		// JavaScript
		// 배열: 대괄호 []
		// 객체: 중괄호 {}

		let arr = [1, 56, 3, 86, 123];
		let obj = { menu: "초밥", price: 15000 };

		// 배열 속에 객체
		// 이 형태로 DB에 들어있는 데이터를 표현한 것: JSON
		let arrobj = [
			{
				menu: "라면",
				price: 4000
			},
			{
				menu: "김밥",
				price: 3000
			},
			{
				menu: "돈까스",
				price: 8000
			}
		];

		// JavaScript의 배열 대상 반복문
		// for (var i = 0; ...) : 인덱스 활용도가 높은 편
		// for (Menu m : arrobj) : 객체에 순서대로 접근이 용이한 편
		// $.each(배열, 데이터 하나 만날 때마다 호출될 콜백함수)
		// 인덱스도 가지고 있고, 객체에 접근하기에도 용이함
		$.each(arrobj, function(i, lunch) {
			alert((i + 1) + ") " + lunch.menu + ": " + lunch.price);
		});

		// DOM 객체 대상 반복문
		// $(선택자).each(DOM객체 하나 만날 때 호출될 콜백함수)

		// h1 => 순서대로 글자의 크기가 10 / 20 / 30pt가 되도록
		$("h1").each(function(i, h) {
			$(h).css("font-size", ((i + 1) * 10) + "pt");
		});
	});
</script>
</head>

<body>
	<h1>ㅋㅋㅋ</h1>
	<h1>ㅎㅎㅎ</h1>
	<h1>ㅠㅠㅠ</h1>
</body>
</html>
```

# 5. Example.html

### 배열 속 객체로 CSS 적용

```jsx
let arrobj = [
	{ color: "green", bgc: "red" },
	{ color: "yellow", bgc: "blue" },
	{ color: "pink", bgc: "teal" }
];
```

```jsx
$("td").each(function(i, t) {
	$(t).css("color", arrobj[i].color);
	$(t).css("background-color", arrobj[i].bgc);
});
```

### 이미지 자동 변경

```jsx
$("img").attr("src", "파일명");
```

- `setInterval()`로 일정 시간마다 이미지 변경
- 마지막 이미지 후 다시 처음 이미지로

### 전체 코드

```html
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>Insert title here</title>
<script type="text/javascript" src="jQuery.js"></script>

<script type="text/javascript">
	$(function() {
		// ㅋㅋㅋㅋㅋ : 글자 색 => green / 배경색 => red
		// ㅎㅎㅎㅎㅎ : 글자 색 => yellow / 배경색 => blue
		// ㅠㅠㅠㅠㅠ : 글자색 => pink / 배경색 => teal

		setInterval(() => {
			i++;

			if (i == 4) {
				i = 1;
			}

			$("img").attr("src", "20260526-" + i + ".png");
		}, 300);
	});
```

# 6. DOM.html

### text() vs html()

```jsx
$("a").text("문자");
$("a").html("<b>문자</b>");
```

- `text()`
    
    → 태그를 문자 그대로 처리
    
- `html()`
    
    → HTML 태그 적용 가능
    

### attr / removeAttr

```jsx
$("a").attr("href", "주소");
$("a").removeAttr("class");
```

- `attr()` : 속성 추가 / 변경 / 가져오기
- `removeAttr()` : 속성 제거

### DOM 객체 생성

```jsx
let m = $("<marquee></marquee>");
```

### 요소 추가

```jsx
$("div").append(m);   // 내부 마지막에 추가
$("div").prepend(m);  // 내부 처음에 추가
$("div").before(m);   // 외부 바로 앞에 추가
$("div").after(m);    // 외부 바로 뒤에 추가
```

### 요소 삭제

```jsx
$("div").remove(); // 자기 자신 + 하위 요소 제거
$("div").empty();  // 하위 요소만 제거
```

### 전체 코드

```html
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>Insert title here</title>
<script type="text/javascript" src="jQuery.js"></script>

<script type="text/javascript">
	$(function() {
		$("#b1").click(function() {
			// a태그로 이루어진 DOM객체 속 text 값 변경
			// $("a").text("유튜브로");

			// text()의 소괄호 안에 값을 넣지 않으면
			// 기존의 값을 가져오게 됨
			let t = $("a").text();
			alert(t);

			// .text()와 비교할 .html()
			// $("a").html("유튜브로");

			// .html()을 사용하면 새로운 html 요소로 변경할 수 있음
			// $("a").text("<b>유튜브로</b>"); // <b> 태그 적용 안 됨
			$("a").html("<b>유튜브로</b>");
		});

		$("#b2").click(function() {
			$("a").attr("href", "https://www.youtube.com");
		});

		// 이 a 태그에 css를 부여하고 싶은데 내용이 많다면
		// id도 부여할 수 있음
		$("a").attr("id", "t");

		// 속성 지우기
		$("a").removeAttr("class"); // class 속성 지우기

		let n = 0;

		$("#b3").click(function() {
			// $("CSS선택자") or $("태그명") : 있는 것을 선택
			// $("DOM객체") : DOM객체 생성

			// marquee DOM객체 추가하기
			n++;

			let m = $("<marquee></marquee>")
						.text("테스트입니다. " + n + "번째");

			// behavior : 스크롤 방식 설정
			// scroll : 한쪽 방향으로만 진행, 기본값
			// alternate : 왕복
			m.attr("behavior", "alternate");

			// direction : 스크롤 방향 설정
			// left 기본값, right, up, down
			m.attr("direction", "right");

			// 요소 추가하기
			// 기존 요소의 내부에 추가 - append, prepend
			$("div").append(m);   // 선택된 요소의 마지막에 내용 추가
			// $("div").prepend(m); // 선택된 요소의 첫 번째에 내용 추가

			// 기존 요소의 외부에 추가 - before, after
```

# 정리

| 개념 | 암기 코드 | 의미 |
| --- | --- | --- |
| 문서 준비 완료 | `$(function() {});` | HTML 로딩 후 실행 |
| 클릭 이벤트 | `$("button").click(function() {});` | 버튼 클릭 시 실행 |
| 값 가져오기 | `$("#i").val();` | input 값 읽기 |
| 값 넣기 | `$("#i").val("값");` | input 값 변경 |
| CSS 변경 | `.css("속성", "값")` | 스타일 변경 |
| 속성 변경 | `.attr("속성", "값")` | href, src 등 변경 |
| 속성 삭제 | `.removeAttr("속성")` | class 등 제거 |
| 텍스트 변경 | `.text("문자")` | 태그 적용 X |
| HTML 변경 | `.html("<b>문자</b>")` | 태그 적용 O |
| 배열 반복 | `$.each(arr, function(i, data) {})` | 배열 순회 |
| DOM 반복 | `$(선택자).each(function(i, dom) {})` | 선택된 요소 순회 |
| 내부 뒤 추가 | `.append()` | 마지막 자식으로 추가 |
| 내부 앞 추가 | `.prepend()` | 첫 자식으로 추가 |
| 외부 앞 추가 | `.before()` | 선택 요소 앞에 추가 |
| 외부 뒤 추가 | `.after()` | 선택 요소 뒤에 추가 |
| 전체 삭제 | `.remove()` | 자기 자신까지 삭제 |
| 내용 삭제 | `.empty()` | 내부 요소만 삭제 |
| 반복 실행 | `setInterval(function, ms)` | 일정 시간마다 반복 |
| 한 번 실행 | `setTimeout(function, ms)` | 일정 시간 후 1회 실행 |
| 랜덤 숫자 | `Math.random()` | 0 이상 1 미만 난수 |
