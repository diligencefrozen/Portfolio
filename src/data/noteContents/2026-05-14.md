function isEmpty(input) { /* 입력값이 비어있는지 확인하는 함수 */
	return !input.value.trim();
}

function isNotNumber(input) { /* 입력값이 숫자가 아닌지 확인하는 함수 */
	return isNaN(input.value);
}

function atLeastLetter(input, len) { /* 입력값의 글자 수가 지정한 길이보다 짧은지 확인하는 함수 */
	return input.value.trim().length < len;
}

function joinCheck() { /*  회원가입 풀을 검사하는 함수 */
	let f = document.joinForm;

	if (isEmpty(f.m_id)) {
		alert("아이디를 입력해주세요.");
		f.m_id.focus();
		return false;
	}

	if (isEmpty(f.m_pw)) {
		alert("비밀번호를 입력해주세요.");
		f.m_pw.focus();
		return false;
	}

	if (isEmpty(f.m_name)) {
		alert("이름을 입력해주세요.");
		f.m_name.focus();
		return false;
	}

	let phoneRegex = /^[0-9]+$/;
	if (isEmpty(f.m_phone) || !phoneRegex.test(f.m_phone.value)) {
		alert("전화번호는 숫자만 입력해주세요.");
		f.m_phone.focus();
		return false;

	if (isEmpty(f.m_img)) {
