package com.beaver.apr022.main;
import java.util.Random;
import java.util.Scanner;

public class Playground {

    public static void main(String[] args) throws InterruptedException {

        Random r = new Random();

        int uNum = 0; // 사용자가 부를 개수
        int cNum = 0; // 컴퓨터가 부를 개수
        int selectNo = 0; // 메뉴 선택 번호
        int win = 0; // 이긴 횟수
        int lose = 0; // 진 횟수

        // [거실] 프로그램이 꺼질 때까지 계속 돌아가는 메인 반복문
        m: while (true) {
            System.out.println("\n--- 베스킨라빈스 31 게임 ---");
            System.out.println("1. 게임 시작 | 2. 게임 전적 | 3. 프로그램 종료");
            System.out.print("메뉴 선택: ");
            selectNo = s.nextInt();

            // 1번 메뉴: 게임 시작 (게임방 입장!)
            if (selectNo == 1) {
                System.out.println(">>> 게임을 시작합니다!");
                int count = 0; // 숫자를 세는 변수 (게임마다 0부터 시작)

                // [게임방] 31이 나올 때까지 반복하는 게임판
                b: while (true) {
                    
                    // ================= 사용자 턴 =================
                    System.out.print("나의 입력 (1~3개): ");
                    uNum = s.nextInt();

                    if (uNum < 1 || uNum > 3) {
                        System.out.println("비겁해요! 1에서 3 사이의 숫자만 입력하세요!");
                        continue;
                    }

                    for (int i = 0; i < uNum; i++) {
                        count++;
                        System.out.println(count + "!");
                        Thread.sleep(500);

                        if (count >= 31) {
                            System.out.println("아이쿠! 31을 불러버렸네요. 패배!");
                            lose++;
                            break b; // 게임방 탈출해서 거실로 이동
                        }
                    }

                    // ================= 컴퓨터 턴 =================
                    System.out.println("컴퓨터가 숫자를 부릅니다...");
                    Thread.sleep(500);
                    
                    if (count >= 27 && count < 30) {
                        cNum = 30 - count;
                    } else {
                        cNum = r.nextInt(3) + 1;
                    }

                    for (int i = 0; i < cNum; i++) {
                        count++;
                        System.out.println(count + "!");
                        Thread.sleep(500);

                        if (count >= 31) {
                            System.out.println("와! 컴퓨터가 31을 불렀어요. 승리!");
                            win++;
                            break b; // 게임방 탈출해서 거실로 이동
                        }
                    }
                } // [게임방 끝]
            } 
            
            // 2번 메뉴: 전적 확인
            else if (selectNo == 2) {
                System.out.println("\n[현재 전적]");
                System.out.println("이긴 횟수: " + win + "번");
                System.out.println("진 횟수: " + lose + "번");
            }

                break m; // 전체 프로그램(거실) 종료

            // 잘못된 메뉴 선택
            else {
                System.out.println("올바른 메뉴 번호를 선택해 주세요!");

        } // [거실 끝]

import java.util.Random;
import java.util.Scanner;

public class PMain2 {

    // 게임을 시작하면서 룰 설명
    public static void printStart(String[] hand) {
        System.out.println("가위바위보 게임 시작");
        System.out.println("######");
        for (int i = 1; i < hand.length; i++) {
            System.out.printf("%d. %s\t\n", i, hand[i]);
        }
        System.out.println("######");
    }

    // 컴퓨터가 뭐 냈는지
    public static int getComHand() {
        Random r = new Random();
        int comHand = r.nextInt(3) + 1;
        return comHand;
    }

    // 유저 입력 (재귀)
    public static int getUserHand() {
        Scanner s = new Scanner(System.in);
        System.out.println("뭐낼래? : ");
        int userHand = s.nextInt();

        if (userHand < 1 || userHand > 3) {
            System.out.println("잘못된 입력!!");
        }

        return (userHand >= 1 && userHand <= 3) ? userHand : getUserHand();
    }

    // 출력
    public static void printHand(int user, int com, String[] h) {
        System.out.println("######");
        System.out.printf("유저\t: %s\t\n", h[user]);
        System.out.printf("컴퓨터\t: %s\t\n", h[com]);
        System.out.println("######");
    }

    // 판정
    public static int judge(int user, int com) {
        int game = user - com;

        if (game == 0) {
            System.out.println("무승부");
            return 0;
        } else if (game == -1 || game == 2) {
            System.out.println("패배");
            return -123123;
        } else {
            System.out.println("승리");
            return 1;
        }
    }

    // 핵심 루프
    public static int getResult(String[] hand) {

        int com = 0;
        int user = 0;
        int win = 0;
        int result = 0;

        while (true) {
            com = getComHand();
            user = getUserHand();

            printHand(user, com, hand);
            result = judge(user, com);

            if (result == -123123) {
                break;
            }

            win += result;
        }

        return win;
    }

    public static void printResult(int win) {
        System.out.println("#####");
        System.out.printf("%d승으로 종료!\t\n", win);
        System.out.println("#####");
    }

    public static void activate(int win) {
        String[] hand = { " ", "가위", "바위", "보" };
        printResult(win);
    }

    public static void main(String[] args) throws InterruptedException {

        // 가위바위보 게임 (함수 사용)
        activate(); // 
    }
}

import java.util.Random;
import java.util.Scanner;

public class PMain3 {

    public static int getNum() {
        Scanner s = new Scanner(System.in);
        int num = s.nextInt();

        if (num < 1 || num > 45) {
            System.out.println("다시입력:");
        }

        return (num >= 1 && num <= 45) ? num : getNum();
    }

    public static int[] getNums() {

        int[] myAnswer = new int[6];

        for (int i = 0; i < myAnswer.length; i++) {

            System.out.printf("%d번째 숫자 입력 : ", i + 1);
            myAnswer[i] = getNum();

            for (int j = 0; j < i; j++) {

                if (myAnswer[i] == myAnswer[j]) {
                    System.out.printf("%번째 숫자와 중복!\n", j + 1);
                    System.out.println("######");
                    i--;
                }
            }
        }

        return myAnswer;
    }

    public static void printMyAnswer(int[] myAnswer) {
        System.out.println("나의 답");

        for (int i : myAnswer) {
            System.out.println("[" + i + "]");
        }

        System.out.println();
        System.out.println("######");
    }

    public static int[] getLotto() {

        int[] lottoNum = new int[6];
        Random r = new Random();

        for (int i = 0; i < lottoNum.length; i++) {

            lottoNum[i] = r.nextInt(45) + 1;

            for (int j = 0; j < i; j++) {

                System.out.println(lottoNum[j] + " " + lottoNum[i]);

                if (lottoNum[i] == lottoNum[j]) {
                    i--;
                    break;
                }
            }
        }

        return lottoNum;
    }

    public static void printLottoNum(int[] lottoNum) {

        System.out.println("로또 번호");

        for (int i : lottoNum) {
            System.out.println("[" + i + "]");
        }

        System.out.println();
        System.out.println("######");
    }

    public static int checkNum(int[] myAnswer, int[] lottoNum) {

        int count = 0;

        for (int i = 0; i < lottoNum.length; i++) {
            for (int j = 0; j < myAnswer.length; j++) {

                if (lottoNum[i] == myAnswer[j]) {
                    count++;
                }
            }
        }

        return count;
    }

    public static void printResult(int count) {

        System.out.printf("%d개 맞춰서...", count);

        if (count == 6) {
            System.out.println("1등");
        } else if (count == 4) {
            System.out.println("2등");
        } else if (count == 3) {
            System.out.println("3등");
        } else {
            System.out.println("꽝");
        }
    }

    public static void activate() {

        int[] lottoNum = getLotto();
        int[] myAnswer = getNums();

        printMyAnswer(myAnswer);
        printLottoNum(lottoNum);

        int count = checkNum(myAnswer, lottoNum);
        printResult(count);
    }

    public static void main(String[] args) {
        activate();
    }
}
