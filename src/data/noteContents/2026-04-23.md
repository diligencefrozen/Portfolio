## 2. Servlet / Tomcat 기본 개념

### Servlet : 자바를 이용해서 웹을 만드는 기술

- WAS(Web Application Server) 에서만 실행 가능
- Tomcat 사용할 것! (1, 80 포트)

### 아파치 톰캣 (Apache Tomcat)

→

자바 기반의 웹 애플리케이션을 실행하는

오픈소스 웹 애플리케이션 서버 (WAS)

자바 서블릿과 JSP 기술을 통해

정적 내용들뿐만 아니라 데이터베이스 연동 등

동적인 웹페이지를 생성하며,

가볍고 유연하며 자바 웹 개발에 가장 많이 사용됨

### 핵심 특징 및 역할 :

- 서블릿(Servlet) 과 JSP(Java Server Pages)를 실행하는 서블릿 컨테이너

→

1. 동적 컨텐츠 처리 : 클라이언트의 요청(HTTP)을 받아 Java의 코드로 실행,
    
    실행 결과물을 HTML로 변환하여 응답
    
2. 독립 실행 : 내장 웹 서버를 통해 별도의 웹 서버 없이 단독 구동 가능
3. 구성요소 : 서블릿 컨테이너인 '카탈리나 (Catalina)',
    
    HTTP 요청을 처리하는 '코요테(Coyote)',
    
    JSP 엔진인 '제스퍼(Jasper)' 로 구성
    
    ⇒ 톰캣 내부에 구성되어있는 라이브러리임!

package com.jeesung.apr232.main;

import java.io.IOException;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

// Servlet

// @XXX : Annotation
//	원래는 직접 다 구현해야 했는데..
// => 자동 처리를 해줌 !
// LearnServlet을 Tomcat에 등록 (*** 지우면 실행이 안됨 !)

@WebServlet("/LearnServlet")
public class LearnServlet extends HttpServlet {
	// HttpServlet 클래스에 있는 기능을 상속 받음
	//			(요청받으면 응답하기)
	// 뭔가를 더 추가 O

	// 작업하면서 버전 표시하기 위해 사용하는 용도 (지워도 무방함)
	private static final long serialVersionUID = 1L;

	// 생성자
	// 별로 쓸일 없음 (지워도 무방함)
    public LearnServlet() {
        super();
    }

    // event-driven-programming (이벤트 기반 프로그래밍)
    // Tomcat이 실행 =>LearnServlet을 실행
    // => tomcat이 LearnServlet 이라는 객체를 만들어서
    //			요청을 기다림
    // => 요청이 오면 해당 URL에 매핑된 서블릿을 찾아 처리(응답)
    // => 대기상태로 전환

    // ** 핵심 부분 !!
    // 대기하고 있다가 클라이언트가
    //	해당 Servlet으로 GET 방식 요청하면
    // Tomcat이 doGET메소드를 자동으로 호출

    // 대기하고 있다가 클라이언트가
    //	해당 Servlet으로 POST 방식 요청하면
    // Tomcat이 doPost메소드를 자동으로 호출
	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		response.getWriter().append("Served at: ").append(request.getContextPath());
	}

	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		doGet(request, response);
	}

}

## 10. Calculator.java

```java
package com.jeesung.apr232.main;

import java.io.IOException;
import java.io.PrintWriter;

import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

/**
 * Servlet implementation class Calculator
 */
@WebServlet("/Calculator")
public class Calculator extends HttpServlet {
	private static final long serialVersionUID = 1L;

    /**
     * @see HttpServlet#HttpServlet()
     */
    public Calculator() {
        super();
        // TODO Auto-generated constructor stub
    }

	/**
	 * @see HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		// TODO Auto-generated method stub
//		response.getWriter().append("Served at: ").append(request.getContextPath());

		response.setCharacterEncoding("UTF-8");
		String x = request.getParameter("x");
		int x1 = Integer.parseInt(request.getParameter("x"));

		String y = request.getParameter("y");
		int y1 = Integer.parseInt(request.getParameter("y"));

		PrintWriter out = response.getWriter();
		out.print("<html>");
		out.print("<head><title>계산기</title>");
		out.print("<meta charset='UTF-8'></head>");
		out.print("<body>");

	    out.print("<h1>사칙연산</h1>");
	    out.print("<table border='1'>");
//		out.print("x+y=");
//		out.print(x1+y1);
		out.printf("<tr><td>%d + %d = %d</td></tr>", x1, y1, x1 + y1);

//		out.print("x-y=");
//		out.print(x1-y1);
		out.printf("<tr><td>%d - %d = %d</td></tr>", x1, y1, x1 - y1);

//		out.print("x*y=");
//		out.print(x1*y1);
		out.printf("<tr><td>%d * %d = %d</td></tr>", x1, y1, x1 * y1);

//		out.print("x/y=");
//		out.print(x1/y1);
		out.printf("<tr><td>%d / %d = %.2f</td></tr>", x1, y1, x1 / (double) y1);

		out.print("</h1>");
		out.print("</table>");
		out.print("</body>");
		out.print("</html>");
	}

	/**
	 * @see HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		// TODO Auto-generated method stub
		doGet(request, response);
	}

}
```

package com.jeesung.apr233.main;

import java.io.IOException;
import java.io.PrintWriter;

import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

/**
 * Servlet implementation class Output
 */

public class Output extends HttpServlet {
	private static final long serialVersionUID = 1L;

    /**
     * @see HttpServlet#HttpServlet()
     */

        super();
        // TODO Auto-generated constructor stub
    }

	/**
	 * @see HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
	 */

 // 사용자가 입력한 데이터가 주소에
 // 인터넷 주소(URL)에 한글 - X
 //   ㅋ => %2B (URL 인코딩)
 // 사용자가 입력한 ㅋ (UTF-8) - URLEncoding -> %2B
 // 사용자로부터 받아온 %2B - URLDecoding -> ㅋ

 // GET 방식 요청 파라미터 한글 처리
//  		UTF-8 로 맞추자!
// 		Tomcat server.xml 65번줄 설정을 건드렸음

	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		// TODO Auto-generated method stub
//		response.getWriter().append("Served at: ").append(request.getContextPath());
		response.setCharacterEncoding("UTF-8");
		String a = request.getParameter("a");
		PrintWriter out = response.getWriter();
		out.print("<html>");
		out.print("<head><title>test</title>");
		out.print("<meta charset='UTF-8'></head>");
		out.print("<body>");
		out.print("<h2>Output-GET</h2>");
		out.printf("<h2>%s</h2>", a);

		out.print("</body>");
		out.print("</html>");

	}

	/**
	 * @see HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
	 */

	// GET vs POST
	// GET : 요청파라미터가 URL에 있음
	//  	대부분 GET 방식을 사용함
	// POST : 요청 파라미터가 주소에 없음, 내부적으로 전달
	//		로그인, 회원가입, ...
	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		// TODO Auto-generated method stub
//		doGet(request, response);
		request.setCharacterEncoding("UTF-8"); // POST 방식에서 이부분을 써주지 않으면 한글값은 깨집니다.
		response.setCharacterEncoding("UTF-8");
		String a = request.getParameter("a");
		PrintWriter out = response.getWriter();
		out.print("<html>");
		out.print("<head><title>test</title>");
		out.print("<meta charset='UTF-8'></head>");
		out.print("<body>");
		out.print("<h2>Output-POST</h2>");
		out.printf("<h2>%s</h2>", a);

		out.print("</body>");
		out.print("</html>");
	}

## 13. GET vs POST 정리

### GET : 요청파라미터가 URL에 있음. 대부분 GET 방식을 사용함

### POST : 요청 파라미터가 주소에 없음, 내부적으로 전달 로그인, 회원가입, ...

### POST 방식에서 이부분을 써주지 않으면 한글값은 깨집니다.

```java
request.setCharacterEncoding("UTF-8");
```

import java.io.IOException;

import javax.servlet.ServletException;

import javax.servlet.http.HttpServlet;

import javax.servlet.http.HttpServletResponse;

/**
 * Servlet implementation class Output
 */
@WebServlet("/Output")
public class Output extends HttpServlet {
	private static final long serialVersionUID = 1L;
       
    /**
     * @see HttpServlet#HttpServlet()
     */
    public Output() {
        super();
        // TODO Auto-generated constructor stub
    }

	/**
	 * @see HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		// TODO Auto-generated method stub
//		response.getWriter().append("Served at: ").append(request.getContextPath());
		response.setCharacterEncoding("UTF-8");
		String id = request.getParameter("id");
		String pw = request.getParameter("pw");
		String gender = request.getParameter("gender");
		String bornselect = request.getParameter("born-select");
		String[] hob = request.getParameterValues("hob");
		String intro = request.getParameter("intro");
		PrintWriter out = response.getWriter();
		out.print("<html>");
		out.print("<head><title>test</title>");
		out.print("<meta charset='UTF-8'></head>");
		out.print("<body>");
		out.print("<h2>Output-GET</h2>");
		out.printf("<h2>%s</h2>", id);
		out.printf("<h2>%s</h2>", pw);
		out.printf("<h2>%s</h2>", gender);
		out.printf("<h2>%s</h2>", bornselect);
		out.printf("<h2>%s</h2>", hob);
		out.printf("<h2>%s</h2>", intro);
		
		
		out.print("</body>");
		out.print("</html>");
	}

	/**
	 * @see HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		// TODO Auto-generated method stub
		doGet(request, response);
	}

}
