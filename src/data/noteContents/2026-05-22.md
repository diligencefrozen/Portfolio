```java
package com.beaver.may212.study;

import java.text.SimpleDateFormat;
import java.util.Date;

import javax.servlet.http.HttpServletRequest;

import org.apache.ibatis.session.SqlSession;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.beaver.may212.study.Study;

@Service
public class StudyDAO {
	
	
	@Autowired
	private SqlSession ss;

	
	public void regStudy(Study c, HttpServletRequest req) {
		try {
			// 과목명 값은 자동 처리 됨 O
			// 날짜 관련 요청파라미터
			// (c_y, c_m, c_d => c_when)
			//		ㄴ자동으로 합쳐주지는 않기에 따로 작업이 필요함.
			String c_y = req.getParameter("c_y");
			// 2026 1 1
			// ㄴ 이 데이터를 불러오면 202611 이 되어버림 => 2026년 11월인지 1월1일인 혼동 발생 -> 그래서 따로 작업이 필요함.
			int c_m =Integer.parseInt(req.getParameter("c_m"));
			int c_d =Integer.parseInt(req.getParameter("c_d"));

			SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMdd");
			
			Date c_when = sdf.parse(c_when2);
```

## StudyMapper.java

```java
package com.beaver.may212.study;

import java.util.List;

public interface StudyMapper {
	// method 리턴타입
	// insert instect / update / delete
	//		=> 영향을 받은 데이터의 수 - int (리턴타입)
	// select
	//		결과가 하나 나온다 : resultType 객체(Student)
	// 		결과가 여러개 나온다 : List<resultType Class> (List<Student>)
	
	// 메소드 명  -mapper.xml의 id 와 맞추기
	// 파라미터 - mapper.xml의 parameterType과 맞추기!
	// classpath:StudentMapper <-- xml 파일 수정해야한다. servlet-context.xml
	// 문장 끝날떄 세미콜론은 작성하지않는다.
	// id 부분에 노란줄 있으면 반드시 수정한다. 다시 작성하거나, 고쳐줘야만 한다.
	public abstract int regStudy(Study c);
	public abstract List<Study> getAllStudy();

	
}
```

## studyMapper.xml

```java
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper
    PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
    "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="com.beaver.may212.study.StudyMapper">

	<insert id="regStudy" parameterType="com.beaver.may212.study.Study">

	    insert into may21_study (
	        c_name,
	        c_when
	    ) values (

	        #{c_name},
	        #{c_when}
	    )
	</insert>

    <select id="getAllStudy" resultType="com.beaver.may212.study.Study">
        select *
        from may21_study
        order by c_when
    </select>

</mapper>
```
