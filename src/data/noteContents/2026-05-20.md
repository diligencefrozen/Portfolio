package com.beaver.test.subway;

import java.io.InputStream;
import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.util.ArrayList;

import javax.servlet.http.HttpServletRequest;

import org.json.simple.JSONArray;
import org.json.simple.JSONObject;
import org.json.simple.parser.JSONParser;

import com.beaver.test.JeesungDBManager;
import com.beaver.test.JeesungHttpClient;

public class SubwayDAO {

    private static final String API_URL =
            "http://openapi.seoul.go.kr:8088/4f626857416f6c6c3632586a416843/json/subwayStationMaster/1/100/";

    public static void printJSONData(HttpServletRequest req) {

        try {
            InputStream is = JeesungHttpClient.download(API_URL);
            String str = JeesungHttpClient.convert(is, "UTF-8");

            JSONParser jp = new JSONParser();
            JSONObject jo = (JSONObject) jp.parse(str);

            JSONObject subwayStationMaster = (JSONObject) jo.get("subwayStationMaster");
            JSONArray row = (JSONArray) subwayStationMaster.get("row");

            for (int i = 0; i < row.size(); i++) {
                JSONObject data = (JSONObject) row.get(i);

                System.out.println(data.get("BLDN_ID"));
                System.out.println(data.get("BLDN_NM"));
                System.out.println(data.get("ROUTE"));
                System.out.println(data.get("LAT"));
                System.out.println(data.get("LOT"));
                System.out.println("----------");
            }

        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    public static void insertData(HttpServletRequest req) {
        Connection con = null;
        PreparedStatement pstmt = null;

        try {
            InputStream is = JeesungHttpClient.download(API_URL);
            String str = JeesungHttpClient.convert(is, "UTF-8");

            JSONParser jp = new JSONParser();
            JSONObject jo = (JSONObject) jp.parse(str);

            JSONObject subwayStationMaster = (JSONObject) jo.get("subwayStationMaster");
            JSONArray row = (JSONArray) subwayStationMaster.get("row");

            con = JeesungDBManager.connect("beaverPool");

            String sql = "insert into jsp_subway "
                       + "(bldn_id, bldn_nm, route, lat, lot) "
                       + "values (?, ?, ?, ?, ?)";

            pstmt = con.prepareStatement(sql);

            for (int i = 0; i < row.size(); i++) {
                JSONObject data = (JSONObject) row.get(i);

                pstmt.setString(1, data.get("BLDN_ID").toString());
                pstmt.setString(2, data.get("BLDN_NM").toString());
                pstmt.setString(3, data.get("ROUTE").toString());
                pstmt.setString(4, data.get("LAT").toString());
                pstmt.setString(5, data.get("LOT").toString());

                pstmt.addBatch();
            }

            pstmt.executeBatch();

            System.out.println("Insert 완료!");

        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            JeesungDBManager.close(con, pstmt, null);
        }
    }

    public static void selectData(HttpServletRequest req) {
        Connection con = null;
        PreparedStatement pstmt = null;
        ResultSet rs = null;

        ArrayList<Subway> subways = new ArrayList<Subway>();

        try {
            con = JeesungDBManager.connect("beaverPool");

            String sql = "select * from jsp_subway";

            pstmt = con.prepareStatement(sql);
            rs = pstmt.executeQuery();

            while (rs.next()) {
                Subway subway = new Subway();

                subway.setBldn_id(rs.getString("bldn_id"));
                subway.setBldn_nm(rs.getString("bldn_nm"));
                subway.setRoute(rs.getString("route"));
                subway.setLat(rs.getString("lat"));
                subway.setLot(rs.getString("lot"));

                subways.add(subway);
            }

            req.setAttribute("subways", subways);

        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            JeesungDBManager.close(con, pstmt, rs);
        }
    }
}

package com.beaver.test;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URL;
import javax.net.ssl.HttpsURLConnection;

public class JeesungHttpClient {

	public static InputStream download(String address) throws IOException {

		URL u = new URL(address);

		if (address.startsWith("https")) {
			HttpsURLConnection huc = (HttpsURLConnection) u.openConnection();
			return huc.getInputStream();
		} else {
			HttpURLConnection huc = (HttpURLConnection) u.openConnection();
			return huc.getInputStream();
		}
	}

	public static String convert(InputStream is, String charset) throws IOException {
		InputStreamReader isr = new InputStreamReader(is, charset);
		BufferedReader br = new BufferedReader(isr);

		String line = null;
		StringBuffer sb = new StringBuffer();

		while ((line = br.readLine()) != null) {
			sb.append(line);
		}
		return sb.toString();
	}
}

package com.jeesung.db.manager;

import java.io.BufferedReader;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.net.URL;
import java.net.URLConnection;
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;

public class JeesungDBManager {

    private static final String ADDR = "jdbc:oracle:thin:@localhost:1521:xe";
    private static final String USER = "test123";
    private static final String PASSWORD = "123456";

    private JeesungDBManager() {
    }

    // Oracle JDBC 드라이버 로딩
    private static void loadOracleDriver() throws SQLException {
        try {
            Class.forName("oracle.jdbc.OracleDriver");
        } catch (ClassNotFoundException e) {
            throw new SQLException("Oracle JDBC Driver를 찾을 수 없습니다. WEB-INF/lib에 ojdbc11.jar가 있는지 확인하세요.", e);
        }
    }

    // 기본 DB 연결
    public static Connection connect() throws SQLException {
        loadOracleDriver();
        return DriverManager.getConnection(ADDR, USER, PASSWORD);
    }

    // 기존 JeesungDBManager.connect("beaverPool") 형태와 맞추기 위한 오버로딩
    public static Connection connect(String poolName) throws SQLException {
        loadOracleDriver();
        return DriverManager.getConnection(ADDR, USER, PASSWORD);
    }

    // API 다운로드
    public static InputStream download(String address) throws Exception {
        URL url = new URL(address);
        URLConnection uc = url.openConnection();

        uc.setConnectTimeout(5000);
        uc.setReadTimeout(5000);

        return uc.getInputStream();
    }

    // InputStream을 문자열로 변환
    public static String convert(InputStream is, String charset) throws Exception {
        StringBuilder sb = new StringBuilder();

        try (
            InputStreamReader isr = new InputStreamReader(is, charset);
            BufferedReader br = new BufferedReader(isr)
        ) {
            String line;

            while ((line = br.readLine()) != null) {
                sb.append(line);
            }
        }

        return sb.toString();
    }

    // select용 close
    public static void close(Connection con, PreparedStatement pstmt, ResultSet rs) {
        close(rs);
        close(pstmt);
        close(con);
    }

    // insert, update, delete용 close
    public static void close(Connection con, PreparedStatement pstmt) {
        close(pstmt);
        close(con);
    }

    // 공통 close
    private static void close(AutoCloseable target) {
        if (target != null) {
            try {
                target.close();
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }
}
