package com.jeesung.may131.board;

import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.util.ArrayList;

import com.jeesung.may131.main.db.JeesungDBManager;

public class BoardDAO {
	public static int write(Board b) {
		String sql = "INSERT INTO board "
				+ "(b_no, b_title, b_content, b_writer, b_date) "
				+ "VALUES (board_seq.nextval, ?, ?, ?, SYSDATE)";
		
		try (
			Connection con = JeesungDBManager.connect();
			PreparedStatement pstmt = con.prepareStatement(sql);
			) {
			pstmt.setString(1, b.getB_title());
			pstmt.setString(2, b.getB_content());
			pstmt.setString(3, b.getB_writer());
			return pstmt.executeUpdate();
		} catch (Exception e) {
			e.printStackTrace();
			return 0;
		}
		
	}
	public static List<Board> getBoards(int page, int pageSize, String keyword) {
		List<Board> boards = new ArrayList<>();
		int start = (page - 1) * pageSize +1;
		int end = page * pageSize;
		boolean hasKeyword = keyword != null && !keyword.trim().equals("");
		String sql;
		
		if (hasKeyword) {
			sql = "SELECT * FROM ( "
				+ "SELECT ROW_NUMBER() OVER (ORDER BY b_no DESC) rn, "
				+ "b_no, b_title, b_content, b_writer, "
				+ "TO_CHAR(b_date, 'YYYY.MM.DD HH24:MI') b_date "
				+ "FROM board "
				+ "WHERE LOWER(b_content) LIKE LOWER(?) "
				+ ") "
				+ "WHERE rn BETWEEN ? AND ?";
			
		} else {
			sql = "SELECT * FROM ( "
				+ "SELECT ROW_NUMBER() OVER (ORDER BY b_no DESC) rn, "
				+ "b_no, b_title, b_content, b_writer, "
				+ "TO_CHAR (b_date, 'YYYY.MM.DD HH24:MI') b_date "
				+ "FROM board "
				+ ") "
				+ "WHERE rn BETWEEN ? AND ?";
		}
		try (
			Connection con = JeesungDBManager.connect();
			PreparedStatement pstmt = con.prepareStatement(sql);
			) {
			if (hasKeyword) {
				pstmt.setString(1, "%" + keyword + "%");
				pstmt.setInt(2, start);
				pstmt.setInt(3, end);
			} else {
				pstmt.setInt(1, start);
				pstmt.setInt(2, end);
			}
			try (ResultSet rs = pstmt.executeQuery()) {
				while (rs.next()) {
					Board b = new Board();

import java.io.IOException;

@WebServlet("/CalcController")
public class CalcController extends HttpServlet {
    private static final long serialVersionUID = 1L;

    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        if (!request.getParameterNames().hasMoreElements()) {
            request.getRequestDispatcher("input.jsp").forward(request, response);
            return;
        }

        int x = Integer.parseInt(request.getParameter("x"));
        int y = Integer.parseInt(request.getParameter("y"));

        int sum = x + y;
        int minus = x - y;
        int multiply = x * y;

        request.setAttribute("y", y);
        request.setAttribute("sum", sum);
        request.setAttribute("minus", minus);
        request.setAttribute("multiply", multiply);
        request.setAttribute("divide", divide);

        request.getRequestDispatcher("result.jsp").forward(request, response);
    }
}

package com.test.bmi;

import java.io.IOException;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.*;

@WebServlet("/BMIController")
public class BMIController extends HttpServlet {
    private static final long serialVersionUID = 1L;

    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        request.getRequestDispatcher("bmiInput.jsp").forward(request, response);
    }

    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        request.setCharacterEncoding("UTF-8");

        String name = request.getParameter("name");
        double height = Double.parseDouble(request.getParameter("height"));
        double weight = Double.parseDouble(request.getParameter("weight"));

        double heightM = height / 100.0;
        double bmi = weight / (heightM * heightM);

        String result;
        if (bmi >= 40) {
            result = "고도비만";
        } else if (bmi >= 35) {
            result = "중증도비만";
        } else if (bmi >= 30) {
            result = "경도비만";
        } else if (bmi >= 25) {
            result = "과체중";
        } else if (bmi >= 18.5) {
            result = "정상";
        } else {
            result = "저체중";
        }

        request.setAttribute("name", name);
        request.setAttribute("height", height);
        request.setAttribute("weight", weight);
        request.setAttribute("bmi", String.format("%.2f", bmi));
        request.setAttribute("result", result);

        request.getRequestDispatcher("bmiResult.jsp").forward(request, response);
    }
}
