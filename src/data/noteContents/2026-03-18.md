# 1. 랜덤 배열 + 최소/최대

```java
import java.util.Random;

public class PMain1 {
    public static void main(String[] args) {

        // 10개의 정수를 랜덤으로 뽑아서 (1~100)
        // 배열에 담고 => 출력

        int[] group = new int[10];
        Random r = new Random();

        // 랜덤 값 저장
        for (int i = 0; i < group.length; i++) {
            group[i] = r.nextInt(100) + 1;
        }

        // 출력
        System.out.println("정수:");
        for (int i : group) {
            System.out.println(i + " ");
        }
        System.out.println();

        // 최소값, 최대값 구하기 (정렬 없이)
        int min = group[0];
        int max = group[0];

        for (int i = 0; i < group.length; i++) {
            if (min > group[i]) {
                min = group[i];
            } else if (max < group[i]) {
                max = group[i];
            }
        }

        System.out.println("최소값 : " + min);
        System.out.println("최대값 :" + max);
    }
}
```

# 2. 학생 점수 프로그램 (절차형)

```java
import java.util.Scanner;

public class PMain2 {
    public static void main(String[] args) {

        Scanner s = new Scanner(System.in);

        int choice = 0;        // 메뉴 선택
        int studentNum = 0;   // 학생 수
        int[] scores = null;  // 점수 배열 (초기 null)

        m: while (true) {

            // 초기 상태 (학생 수 먼저 입력해야 함)
            if (scores == null) {
                System.out.println("1.학생수\n5. 종료");
                System.out.println("선택>");
                choice = s.nextInt();

                if (!(choice == 1 || choice == 5)) {
                    System.out.println("잘못된 메뉴 선택!");
                    continue;
                }
            } else {
                System.out.println("1. 학생 수 값을 입력");
                System.out.println("2. 각 학생들의 점수를 입력(0~100)");
                System.out.println("3. 점수 리스트 출력");
                System.out.println("4. 최고/평균 출력");
                System.out.println("5. 종료");
                System.out.println("선택>");
                choice = s.nextInt();
            }

            switch (choice) {

                case 1:
                    System.out.println("학생 수:");
                    studentNum = s.nextInt();

                    if (studentNum <= 0) {
                        System.out.println("1명 이상!");
                        break;
                    }

                    scores = new int[studentNum];
                    break;

                case 2:
                    for (int i = 0; i < scores.length; i++) {
                        System.out.printf("%d번 학생 점수:", i + 1);
                        scores[i] = s.nextInt();

                        // 잘못 입력 시 다시 입력
                        if (scores[i] < 0 || scores[i] > 100) {
                            System.out.println("잘못된 점수!");
                            i--; // 🔥 핵심
                        }
                    }
                    break;

                case 3:
                    int index = 1;
                    for (int i : scores) {
                        System.out.printf("%d번 학생 점수 : %d\n", index++, i);
                    }
                    break;

                case 4:
                    int max = 0;
                    int sum = 0;
                    double avg = 0;

                    for (int i : scores) {
                        sum += i;
                        max = (max < i) ? i : max;
                    }

                    avg = (double) sum / scores.length;

                    System.out.println("최고 점수 : " + max);
                    System.out.println("평균 점수 : " + avg);
                    break;

                case 5:
                    System.out.println("종료");
                    break m;

                default:
                    System.out.println("잘못 입력");
            }
        }
    }
}
```
