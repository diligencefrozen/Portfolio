package com.beaver.may212.student;

import java.util.List;

import javax.servlet.http.HttpServletRequest;

import org.apache.ibatis.session.SqlSession;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.beaver.may212.study.Study;

@Service
public class StudentDAO {
	
	
	@Autowired
	private SqlSession ss;
	
	
	public void regStudent(Student s, HttpServletRequest req) {
		try {
			// 연결 - 자동 sqlsession 을 불러옴으로서...(connect, close 필요 x)
			
			// 값 받아오기 & 객체를 만들기
			// Spring이 해결을 자동으로 해줌
			
			// 내용출력
			System.out.println(s.getS_name());
			System.out.println(s.getS_nickname());
			
			// SQL 수행 - MyBatis 활용
			// => interface를 가져온다!
			StudentMapper sm = ss.getMapper(StudentMapper.class);
			
			// interface에 있는 abstract class (추상 메소드) 호출
			// -자동 -> 연결이 되어있는 mapper.xml의 해당 부분이 실행
			if (sm.regStudent(s)==1) {
				req.setAttribute("r", "학생 등록 성공");
//				ss.commit(); => commit 자동 !!
			}
		} catch (Exception e) {
			// TODO: handle exception
			e.printStackTrace();
			req.setAttribute("r", "학생 등록 실패");
		}
	}
	public void getAllStudent(HttpServletRequest req) {
		try {
			List<Student> students =
			ss.getMapper(StudentMapper.class).getAllStudent();
			req.setAttribute("students", students);
		} catch 
		(Exception e) {
			// TODO: handle exception
			e.printStackTrace();
		}
	}
	
	public void regStudy(Study st, HttpServletRequest req) {
		try {
			System.out.println(st.getC_name());
			System.out.println(st.getC_y());
			System.out.println(st.getC_m());
			System.out.println(st.getC_d());
			
			
		} catch (Exception e) {
			// TODO: handle exception
			e.printStackTrace();
		}
	}

}
