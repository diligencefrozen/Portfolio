		Connection con = null;
		PreparedStatement pstmt = null;
		
		try {
			con = BeaverDBManager.connect();
			String sql =  "insert into reservation "
					+"values(reservation_seq.nextval, "
					+"?, ?, ?, ?)";
			pstmt = con.prepareStatement(sql);
			pstmt.setString(1,rsv.getName());
			pstmt.setDate(2, new Date(rsv.getWhen().getTime()));
			pstmt.setString(3, rsv.getPhone());
			pstmt.setString(4, rsv.getLocation());
			
			if (pstmt.executeUpdate() == 1) {
				return "예약 성공";
			}
			return ""; // Java문법상 필요해서 억지로 넣은 것 !
		} catch (Exception e) {
			// TODO: handle exception
			e.printStackTrace();
			return "예약 실패";
		} finally {
			BeaverDBManager.close(con, pstmt, null);
		}

# 2. 두개의 테이블 : market / product

## 2-1. 만들기

```sql
drop table product cascade constraints purge;
drop table market cascade constraints purge;
drop sequence product_seq;
drop sequence market_seq;

create table market(
    m_no number(4) primary key,
    m_name varchar2(20 char) not null,
    m_location varchar2(20 char) not null,
    m_land number(5) not null,
    m_parking number(5) not null
);

create sequence market_seq start with 1 increment by 1;

create table product(
    p_no number(4) primary key,
    p_name varchar2(20 char) not null,
    p_weight number(5) not null,
    p_price number(6) not null,
    p_m_no number(4) not null,
    constraint fk_product_market
        foreign key(p_m_no)
        references market(m_no)
        on delete cascade
);

create sequence product_seq start with 1 increment by 1;
```

`market`은 마트 정보, `product`는 상품 정보.

`p_m_no`는 상품이 어느 마트 소속인지 저장하는 외래키다.

# 4. rownum / inline view

`rownum`은 정렬보다 먼저 붙기 때문에, **정렬 후 몇 번째**를 구하려면 `inline view`가 필요하다.

```sql
select rownum, m_no, m_name, m_price
from (
    select m_no, m_name, m_price
    from menu
    order by m_name asc
)
where rownum <= 3;
```

이름순 정렬 후 상위 3개.

```sql
select *
from (
    select rownum as rn, m_no, m_name, m_price
    from (
        select m_no, m_name, m_price
        from menu
        order by m_name asc
    )
)
where rn between 2 and 5;
```

이름순 정렬 후 2~5번째.

```sql
select *
from (
    select rownum as rn, m_name, m_price
    from (
        select m_name, m_price
        from menu
        order by m_price desc, m_name asc
    )
)
where rn between 3 and 6;
```

가격 내림차순, 같으면 이름 오름차순, 3~6번째.
